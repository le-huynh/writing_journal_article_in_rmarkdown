\begin{equation}
\label{eq:cutoff}
p(x) = P(Y = 1|X = x) = 
\left\{
    \begin{array}{lr}
          p_1 = P(Y = 1|X \le cp), & \text{if } x \le cp\\
          p_2 = P(Y = 1|X > cp), & \text{if } x > cp
    \end{array}
\right.
\end{equation}

\begin{subequations}
  \label{eq:model}
  \begin{align}
  \label{eq:modela}
P(y|\theta, \lambda) = 
\left\{
    \begin{array}{lr}
          \theta & \text{if } y = 0\\
          (1 - \theta) \frac{Poisson(y|\lambda)}{1 - PoissonCDF(0|\lambda)} 
          & \text{if } y > 0
    \end{array}
\right. \\
  \label{eq:modelb}
logit(\theta) = \alpha_0 + \alpha_1 * x_1 + \alpha_2 * x_2 \\
  \label{eq:modelc}
log(\lambda) = \beta_0 + \beta_1 * x_1 + \beta_2 * x_2 + \nu
  \end{align}
\end{subequations}

  \texttt{@Davis2009}: cite directly Davis et al. (2009)
\item
